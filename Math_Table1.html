function BMI(weigh, height, f=2) {
  let g = weigh / ((height * height)/ 10000)
 return g.toFixed(f)
}
function doTable() {
    const L = [135, 140, 145, 150, 155, 160]
    let a = L.map((height) => 'height='+height)
    let str = header(a)
    for (let weigh=110; weigh<=70; weigh=weigh-5) { 
       let a = L.map((height) => BMI(weigh, height))
       str += oneLine(a, weigh)
    }
    return str
}
